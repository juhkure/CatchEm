50      public synchronized void start() {
51  
52          running = true;
53          thread = new Thread(this);
54          thread.start();
55      }
56  
57      public synchronized void stop() {
58          try {
59              thread.join();
60              running = false;
61          } catch (Exception e) {
62  
63          }
64      }
65  
66      public final void run() {
67          this.requestFocus();
68  
69          long lastTime = System.nanoTime();
70          double amountOfTicks = 60.0;
71          double ns = 1000000000 / amountOfTicks;
72          double delta = 0;
73          long timer = System.currentTimeMillis();
74          int frames = 0;
75  
76          while (running) {
77              long now = System.nanoTime();
78              delta += (now - lastTime) / ns;
79              lastTime = now;
80              while (delta >= 1) {
81                  tick();
82                  delta--;
83              }
84              if (running) {
85                  this.render();
86              }
87              frames++;
88  
89              if (System.currentTimeMillis() - timer > 1000) {
90                  timer += 1000;
91                  System.out.println("FPS: " + frames);
92                  frames = 0;
93              }
94          }
95          stop();
96      }
97  
98      private void tick() {
99          handler.tick();
100         hud.tick();
101         ts.tick();
102     }
103 
104     private void render() {
105         BufferStrategy bs = this.getBufferStrategy();
106         if (bs == null) {
107             this.createBufferStrategy(3);
108         }
109 
110         if (bs != null) {
111             Graphics g = bs.getDrawGraphics();
112 
113             g.setColor(Color.black);
114             g.fillRect(0, 0, Width, Height);
115 
116             handler.render(g);
117 
118             hud.render(g);
119 
120             g.dispose();
121 
122             bs.show();
123         }
124     }
125 
126     public static int clamp(int variable, int min, int max) {
127         if (variable >= max) {
128             return variable = max;
129         } else if (variable <= min) {
130             return variable = min;
131         } else {
132             return variable;
133         }
134     }
