6   package CatchEm;
7   
8   import static CatchEm.Game.Height;
9   import static CatchEm.Game.Width;
10  import java.awt.Color;
11  import java.awt.Graphics;
12  import java.awt.Paint;
13  import java.awt.Rectangle;
14  import java.util.Random;
15  
16  /**
17   *
18   * @author Johannes
19   */
20  public class Player extends GameObject {
21  
22      private Random random;
23      private Handler handler;
24      private HUD hud;
25      private Spawner spawner;
26  
27      /**
28       * Alustetaan pelaaja
29       *
30       * @param x x -koodrinaatti
31       * @param y y -koordinaatti
32       * @param id ID tunnistautumista varten
33       * @param handler
34       * @param hud
35       * @param spawner
36       */
37      public Player(int x, int y, ID id, Handler handler, HUD hud, Spawner spawner) {
38          super(x, y, id);
39          random = new Random();
40          this.score = 0;
41          this.handler = handler;
42          this.hud = hud;
43          this.stamina = 100;
44          this.spawner = spawner;
45      }
46  
47      @Override
48      public void tick() {
49          x += velX;
50          y += velY;
51  
52          x = Game.clamp(x, 0, Game.Width - 39);
53          y = Game.clamp(y, 0, Game.Height - 60);
54  
55          collision();
56      }
57      
58      /**
59       * Joka peli tickin aikana collision metodi tarkistaa mikäli pelaajan ja 
60       * targetin välillä on "fyysinen" yhteys.
61       */
62  
63      private void collision() {
64          for (int i = 0; i < handler.object.size(); i++) {
65  
66              GameObject tempObject = handler.object.get(i);
67  
68              if (tempObject.getId() == ID.Target) {
69                  if (getBounds().intersects(tempObject.getBounds())) {
70                      this.addScore();
71                      handler.removeObject(tempObject);
72                      spawner.addTarget();
73                      hud.addTime();
74                  }
75              }
76          }
77      }
78  
79      /**
80       *
81       * @param g
82       */
83      @Override
84      public void render(Graphics g) {
85          if (getId() == ID.Player) {
86              g.setColor(Color.blue);
87  
88              g.fillRect(x, y, 32, 32);
89          }
90          if (getId() == ID.Player2) {
91              g.setColor(Color.green);
92              g.fillRect(x, y, 32, 32);
93          }
94      }
95  
96      /**
97       *
98       * @return
99       */
100     @Override
101     public Rectangle getBounds() {
102         return new Rectangle(x, y, 32, 32);
103     }
104 
105 }
